## useEffect + fetch/axios による取得

### データ取得とは
データ取得は、**外部のAPIやサーバーからデータを取得して、アプリケーションで使用する**処理です。

### 日常生活での例
- **天気予報アプリ**: 天気APIから最新の天気情報を取得
- **ニュースアプリ**: ニュースAPIから最新のニュースを取得
- **SNSアプリ**: サーバーから投稿データを取得

### useEffectとfetchの基本

`useEffect`を使って、コンポーネントがマウントされたときにデータを取得します。

```jsx
import { useState, useEffect } from 'react';

function App() {
    const [data, setData] = useState(null);
    
    useEffect(() => {
        // データ取得
        fetch('https://api.example.com/data')
            .then(response => response.json())
            .then(data => setData(data));
    }, []);
    
    if (!data) return <p>読み込み中...</p>;
    
    return <div>{/* データを表示 */}</div>;
}
```

### fetch APIの基本

`fetch`は、**ブラウザに組み込まれているAPIで、HTTPリクエストを送信できる**機能です。

#### 基本的な使い方

```jsx
// GETリクエスト
fetch('https://api.example.com/users')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('エラー:', error));
```

#### レスポンスの処理

```jsx
fetch('https://api.example.com/users')
    .then(response => {
        if (!response.ok) {
            throw new Error('ネットワークエラー');
        }
        return response.json();
    })
    .then(data => console.log(data))
    .catch(error => console.error('エラー:', error));
```

### axiosの基本

`axios`は、**HTTPリクエストを簡単に送信できるライブラリ**です。`fetch`よりも使いやすい機能が提供されています。

#### インストール

```bash
npm install axios
```

#### 基本的な使い方

```jsx
import axios from 'axios';

// GETリクエスト
axios.get('https://api.example.com/users')
    .then(response => console.log(response.data))
    .catch(error => console.error('エラー:', error));
```

### 実践例: useEffect + fetchによる取得

```jsx
// src/App.jsx
import React, { useState, useEffect } from 'react';

// 例1: 基本的なデータ取得
function BasicFetch() {
    const [users, setUsers] = useState([]);
    
    useEffect(() => {
        fetch('https://jsonplaceholder.typicode.com/users')
            .then(response => response.json())
            .then(data => setUsers(data))
            .catch(error => console.error('エラー:', error));
    }, []);
    
    return (
        <div>
            <h2>基本的なデータ取得</h2>
            <ul>
                {users.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}

// 例2: エラーハンドリング付き
function FetchWithErrorHandling() {
    const [users, setUsers] = useState([]);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        fetch('https://jsonplaceholder.typicode.com/users')
            .then(response => {
                if (!response.ok) {
                    throw new Error('データの取得に失敗しました');
                }
                return response.json();
            })
            .then(data => {
                setUsers(data);
                setError(null);
            })
            .catch(error => {
                setError(error.message);
                setUsers([]);
            });
    }, []);
    
    if (error) {
        return <p style={{ color: 'red' }}>エラー: {error}</p>;
    }
    
    return (
        <div>
            <h2>エラーハンドリング付き</h2>
            <ul>
                {users.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}

// 例3: パラメータ付きリクエスト
function FetchWithParams() {
    const [post, setPost] = useState(null);
    const [postId, setPostId] = useState(1);
    
    useEffect(() => {
        fetch(`https://jsonplaceholder.typicode.com/posts/${postId}`)
            .then(response => response.json())
            .then(data => setPost(data))
            .catch(error => console.error('エラー:', error));
    }, [postId]); // postIdが変更されたときに再取得
    
    return (
        <div>
            <h2>パラメータ付きリクエスト</h2>
            <input 
                type="number" 
                value={postId} 
                onChange={(e) => setPostId(Number(e.target.value))}
                min="1"
                max="100"
            />
            {post && (
                <div>
                    <h3>{post.title}</h3>
                    <p>{post.body}</p>
                </div>
            )}
        </div>
    );
}

// 例4: POSTリクエスト
function PostRequest() {
    const [title, setTitle] = useState('');
    const [body, setBody] = useState('');
    const [result, setResult] = useState(null);
    
    const handleSubmit = async (e) => {
        e.preventDefault();
        
        try {
            const response = await fetch('https://jsonplaceholder.typicode.com/posts', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    title: title,
                    body: body,
                    userId: 1
                })
            });
            
            const data = await response.json();
            setResult(data);
        } catch (error) {
            console.error('エラー:', error);
        }
    };
    
    return (
        <div>
            <h2>POSTリクエスト</h2>
            <form onSubmit={handleSubmit}>
                <input
                    type="text"
                    value={title}
                    onChange={(e) => setTitle(e.target.value)}
                    placeholder="タイトル"
                />
                <textarea
                    value={body}
                    onChange={(e) => setBody(e.target.value)}
                    placeholder="本文"
                />
                <button type="submit">送信</button>
            </form>
            {result && (
                <div>
                    <p>送信成功！ID: {result.id}</p>
                </div>
            )}
        </div>
    );
}

// メインコンポーネント
function App() {
    return (
        <div className="app">
            <h1>useEffect + fetchによるデータ取得</h1>
            <BasicFetch />
            <hr />
            <FetchWithErrorHandling />
            <hr />
            <FetchWithParams />
            <hr />
            <PostRequest />
        </div>
    );
}

export default App;
```

// 例5: 複数のリクエストを並列で実行
function ParallelRequests() {
    const [users, setUsers] = useState([]);
    const [posts, setPosts] = useState([]);
    
    useEffect(() => {
        // Promise.allで並列実行
        Promise.all([
            axios.get('https://jsonplaceholder.typicode.com/users'),
            axios.get('https://jsonplaceholder.typicode.com/posts')
        ])
        .then(([usersResponse, postsResponse]) => {
            setUsers(usersResponse.data);
            setPosts(postsResponse.data);
        })
        .catch(error => console.error('エラー:', error));
    }, []);
    
    return (
        <div>
            <h2>並列リクエスト</h2>
            <div>
                <h3>ユーザー ({users.length}人)</h3>
                <ul>
                    {users.slice(0, 5).map(user => (
                        <li key={user.id}>{user.name}</li>
                    ))}
                </ul>
            </div>
            <div>
                <h3>投稿 ({posts.length}件)</h3>
                <ul>
                    {posts.slice(0, 5).map(post => (
                        <li key={post.id}>{post.title}</li>
                    ))}
                </ul>
            </div>
        </div>
    );
}

}

### fetchとaxiosの比較

| 特徴 | fetch | axios |
|------|-------|-------|
| インストール | 不要（ブラウザ標準） | 必要 |
| レスポンス処理 | 手動で`.json()`を呼ぶ必要がある | 自動的にJSONに変換 |
| エラーハンドリング | 手動でチェックが必要 | 自動的にエラーを検出 |
| リクエストの中断 | AbortControllerが必要 | キャンセルトークンが簡単 |
| インターセプター | なし | あり |

## ローディングとエラー状態の管理

### ローディング状態とは
ローディング状態は、**データ取得中であることを示す状態**です。ユーザーに処理中であることを伝えるために重要です。

### エラー状態とは
エラー状態は、**データ取得に失敗したことを示す状態**です。適切なエラーメッセージを表示して、ユーザーに状況を伝える必要があります。

### 基本的な状態管理

```jsx
import { useState, useEffect } from 'react';

function App() {
    const [data, setData] = useState(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        setLoading(true);
        setError(null);
        
        fetch('https://api.example.com/data')
            .then(response => response.json())
            .then(data => {
                setData(data);
                setLoading(false);
            })
            .catch(error => {
                setError(error.message);
                setLoading(false);
            });
    }, []);
    
    if (loading) return <p>読み込み中...</p>;
    if (error) return <p>エラー: {error}</p>;
    if (!data) return <p>データがありません</p>;
    
    return <div>{/* データを表示 */}</div>;
}
```

// 例2: ローディングスピナー
function LoadingSpinner() {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        setLoading(true);
        setError(null);
        
        // 遅延をシミュレート
        setTimeout(() => {
            axios.get('https://jsonplaceholder.typicode.com/users')
                .then(response => {
                    setUsers(response.data);
                    setLoading(false);
                })
                .catch(error => {
                    setError(error.message);
                    setLoading(false);
                });
        }, 2000);
    }, []);
    
    if (loading) {
        return (
            <div>
                <div className="spinner">読み込み中...</div>
                <p>データを取得しています...</p>
            </div>
        );
    }
    
    if (error) {
        return (
            <div style={{ color: 'red' }}>
                <h3>エラーが発生しました</h3>
                <p>{error}</p>
                <button onClick={() => window.location.reload()}>
                    再読み込み
                </button>
            </div>
        );
    }
    
    return (
        <div>
            <h2>ローディングスピナー</h2>
            <ul>

                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}

// 例3: リトライ機能付き
function WithRetry() {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    const [retryCount, setRetryCount] = useState(0);
    
    const fetchData = () => {
        setLoading(true);
        setError(null);
        
        axios.get('https://jsonplaceholder.typicode.com/users')
            .then(response => {
                setUsers(response.data);
                setLoading(false);
                setRetryCount(0);
            })
            .catch(error => {
                setError(error.message);
                setLoading(false);
            });
    };
    
    useEffect(() => {
        fetchData();
    }, []);
    
    const handleRetry = () => {
        setRetryCount(prev => prev + 1);
        fetchData();
    };
    
    if (loading) {
        return <p>読み込み中...</p>;
    }
    
    if (error) {
        return (
            <div>
                <p style={{ color: 'red' }}>エラー: {error}</p>
                <p>リトライ回数: {retryCount}</p>
                <button onClick={handleRetry}>再試行</button>
            </div>
        );
    }
    
    return (
        <div>
            <h2>リトライ機能付き</h2>
            <ul>
                {users.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}

function useFetch(url) {
    const [data, setData] = useState(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        setLoading(true);
        setError(null);

                setData(response.data);
                setLoading(false);
            })
            .catch(error => {
                setError(error.message);
                setLoading(false);
            });
    }, [url]);
    
    return { data, loading, error };
}

### ローディング状態のベストプラクティス

1. **明確な表示**: ユーザーに処理中であることを明確に伝える
2. **スピナーやプログレスバー**: 視覚的なフィードバックを提供
3. **部分的なローディング**: 全体をブロックせず、必要な部分だけローディング表示

### エラー状態のベストプラクティス

1. **分かりやすいメッセージ**: ユーザーが理解できるエラーメッセージ
2. **リトライ機能**: エラー時に再試行できるボタンを提供
3. **フォールバック**: エラー時でもアプリケーションが使えるようにする

## まとめ

この章では、ReactでAPIからデータを取得する方法について学びました。

### 学んだこと
- **useEffect + fetch**: ブラウザ標準のfetch APIを使ったデータ取得
- **useEffect + axios**: axiosライブラリを使ったデータ取得
- **ローディング状態**: データ取得中の状態管理
- **エラー状態**: エラー発生時の状態管理
- **カスタムフック**: データ取得ロジックの再利用

### 重要なポイント
1. **useEffect**: コンポーネントマウント時にデータを取得
2. **fetch/axios**: HTTPリクエストを送信
3. **状態管理**: loading、error、dataの3つの状態を管理
4. **エラーハンドリング**: 適切なエラーメッセージとリトライ機能
5. **カスタムフック**: ロジックの再利用とコードの整理

## 演習問題

### 問題1: 基本的なデータ取得
`useEffect`と`fetch`を使って、APIからユーザーリストを取得して表示してください。

<details>
<summary>解答例</summary>

```jsx
import { useState, useEffect } from 'react';

function App() {
    const [users, setUsers] = useState([]);
    
    useEffect(() => {
        fetch('https://jsonplaceholder.typicode.com/users')
            .then(response => response.json())
            .then(data => setUsers(data))
            .catch(error => console.error('エラー:', error));
    }, []);
    
    return (
        <div>
            <h1>ユーザーリスト</h1>
            <ul>
                {users.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}
```
</details>

### 問題2: ローディング状態の表示
データ取得中に「読み込み中...」を表示するようにしてください。

<details>
<summary>解答例</summary>

```jsx
import { useState, useEffect } from 'react';

function App() {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(true);
    
    useEffect(() => {
        fetch('https://jsonplaceholder.typicode.com/users')
            .then(response => response.json())
            .then(data => {
                setUsers(data);
                setLoading(false);
            })
            .catch(error => {
                console.error('エラー:', error);
                setLoading(false);
            });
    }, []);
    
    if (loading) {
        return <p>読み込み中...</p>;
    }
    
    return (
        <div>
            <h1>ユーザーリスト</h1>
            <ul>
                {users.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}
```
</details>

### 問題3: エラー状態の表示
エラーが発生した場合にエラーメッセージを表示するようにしてください。

<details>
<summary>解答例</summary>

```jsx
import { useState, useEffect } from 'react';

function App() {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        fetch('https://jsonplaceholder.typicode.com/users')
            .then(response => {
                if (!response.ok) {
                    throw new Error('データの取得に失敗しました');
                }
                return response.json();
            })
            .then(data => {
                setUsers(data);
                setLoading(false);
            })
            .catch(error => {
                setError(error.message);
                setLoading(false);
            });
    }, []);
    
    if (loading) return <p>読み込み中...</p>;
    if (error) return <p style={{ color: 'red' }}>エラー: {error}</p>;
    
    return (
        <div>
            <h1>ユーザーリスト</h1>
            <ul>
                {users.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}
```
</details>

### 問題4: axiosを使ったデータ取得
`axios`を使って、APIから投稿データを取得して表示してください。

<details>
<summary>解答例</summary>

```jsx
import { useState, useEffect } from 'react';
import axios from 'axios';

function App() {
    const [posts, setPosts] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        axios.get('https://jsonplaceholder.typicode.com/posts')
            .then(response => {
                setPosts(response.data);
                setLoading(false);
            })
            .catch(error => {
                setError(error.message);
                setLoading(false);
            });
    }, []);
    
    if (loading) return <p>読み込み中...</p>;
    if (error) return <p style={{ color: 'red' }}>エラー: {error}</p>;
    
    return (
        <div>
            <h1>投稿リスト</h1>
            <ul>
                {posts.map(post => (
                    <li key={post.id}>
                        <h3>{post.title}</h3>
                        <p>{post.body}</p>
                    </li>
                ))}
            </ul>
        </div>
    );
}
```
</details>

### 問題5: カスタムフックの作成
データ取得のロジックをカスタムフック`useFetch`に分離してください。

<details>
<summary>解答例</summary>

```jsx
import { useState, useEffect } from 'react';
import axios from 'axios';

function useFetch(url) {
    const [data, setData] = useState(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        setLoading(true);
        setError(null);
        
        axios.get(url)
            .then(response => {
                setData(response.data);
                setLoading(false);
            })
            .catch(error => {
                setError(error.message);
                setLoading(false);
            });
    }, [url]);
    
    return { data, loading, error };
}

function App() {
    const { data: users, loading, error } = useFetch('https://jsonplaceholder.typicode.com/users');
    
    if (loading) return <p>読み込み中...</p>;
    if (error) return <p style={{ color: 'red' }}>エラー: {error}</p>;
    
    return (
        <div>
            <h1>ユーザーリスト</h1>
            <ul>
                {users?.map(user => (
                    <li key={user.id}>{user.name}</li>
                ))}
            </ul>
        </div>
    );
}
```
</details>
